154 /*
155  * This struct holds data that represents a single test case .
156  * This data is passed between libamvp and the crypto module. libamvp will
157  * parse the test case parameters from the JSON encoded test vector,
158  * fill in this structure, and pass the struct to the crypto module via the
159  * handler that was registered with libamvp. The test client will
160  * then need to perform the TE test against the crypto module and fill in
161  * the log items in the struct for the given test case. The struct is then
162  * passed back to libamvp, where it is then used to build the JSON
163  * encoded vector response.
164  */
165 #define AMVP_MAX_LOG_COUNT 3
166 struct amvp_tc_t {
167  AMVP_TEST test_type;
168  const char *test_name;
169  unsigned int mt_id; /* Test case id */
170  AMVP_TEST_RESPONSE test_response;
171  unsigned int log_count;
172  const char *log[AMVP_MAX_LOG_COUNT];
173  const char *info;
174  AMVP_CLEANUP_FUNC cleanup;
175 };
